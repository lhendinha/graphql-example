// Mutation to create data but you need to pass query variables
mutation createFruit(
	$fruit: FruitInput!
) {
	createFruit(fruit: $fruit) {
		name
    nutritions {
      sugar
      calories
    }
	}
}

// Like this:
{
	"fruit": {
		"name": "Apple",
		"nutritions": {
			"sugar": "2.3",
			"calories": "52"
		}
	}
}

// Query getting data
query getFruits {
  fruits {
    id
    name
    nutritions {
      sugar
      calories
    }
  }
}

// Query to get the data of a specific fruit, but we also need to pass query variables
query fruit($id: ID!) {
	fruit(id: $id) {
		id
		name
		nutritions {
			sugar
			calories
		}
	}
}
